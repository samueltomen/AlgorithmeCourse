algorithme exerciceBoucle4
variable
    resultat:integer
    nombre:integer
debut
    saisir nombre
    resultat <- 1
    pour i de 1 à nombre pas de 1 faire
        resultat <- resultat * i
    finfaire
    afficher resultat
fin
